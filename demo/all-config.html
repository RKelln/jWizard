Code

$("#jWizard").jWizard({
   validate: false,   // [Dependency: jQuery Validation Plugin] (http://bassistance.de/jquery-plugins/jquery-plugin-validation/)
   debug: false,   // Enables internal logging (Not fully implemented)

   titleHide: false,   // Determines whether or not to generate/use a title bar
   menuEnable: false,   // Determines whether or not to generate/use a menu

   buttons: {
      cancelHide: false,   // Determines whether or not to generate/use a cancel <button>
      cancelType: "button",   // Determines the `type` attribute on the cancel <button> (http://www.w3schools.com/tags/att_button_type.asp)
      finishType: "button",   // Determines the `type` attribute on the finish <button> (http://www.w3schools.com/tags/att_button_type.asp)
      cancelText: "Cancel",   // Determines the text of the cancel <button>
      previousText: "Previous",   // Determines the text of the previous <button>
      nextText: "Next",   // Determines the text of the next <button>
      finishText: "Finish"   // Determines the text of the finish <button>
   },

   counter: {
      enable: false,   // Determines whether or not to generate/use a counter
      type: "count",   // Determines the type of counter (possible values: "count", "percentage")
      progressbar: false,   // [Dependency: jQuery UI ProgressBar] -- If true: Sets the background of the counter as a jQuery UI ProgressBar
      location: "footer",   // Determines where the counter is to be placed in the wizard (only "header" & "footer" supported)
      startCount: true,   // Determines whether or not to include the first step in determining progress
      startHide: false,   // Determines whether or not to show the counter on the first step
      finishCount: true,   // Determines whether or not to include the last step in determining progress
      finishHide: false,   // Determines whether or not to show the counter on the last step
      appendText: "Complete",   // The "appended text" of the counter (ie. 25% Complete)
      orientText: "left"   // Determines how to orient the text of the counter (possible values: left, center, right)
   },

   effects: {
      enable: false,   // [Dependency: jQuery UI Effects] -- Determines whether or not to animate the Wizard
      step: {
         enable: true,   // Determines whether or not to animate the changing of steps

         // This is the jQuery UI Effect to be performed when hiding the current step
         hide: {
            type: "slide",
            options: {
               direction: "left"
            },
            duration: "normal",
            callback: $.noop
         },
         // This is the jQuery UI Effect to be performed when showing the next step
         show: {
            type: "slide",
            options: {
               direction: "left"
            },
            duration: "normal",
            callback: $.noop
         },
      },
      title: {
         enable: true,   // Determines whether or not to animate title during step changes

         // This is the jQuery UI Effect to be performed when hiding the current step's title
         hide: {
            type: "slide",
            options: {},
            duration: "normal",
            callback: $.noop
         },
         // This is the jQuery UI Effect to be performed when showing the next step's title
         show: {
            type: "slide",
            options: {},
            duration: "normal",
            callback: $.noop
         }
      },
      menu: {
         enable: true,   // Determines whether or not to animate the menu after a step change

         // This is the jQuery UI Effect to be performed on the next step's entry in the menu after the step is changed
         change: {
            type: "highlight",
            options: {},
            duration: "normal",
            callback: $.noop
         }
      },
      counter: {
         enable: true,   // Determines whether or not to animate the counter after a step change

         // This is the jQuery UI Effect to be performed on the counter after the step is changed
         change: {
            type: "highlight",
            options: {},
            duration: "normal",
            callback: $.noop
         }
      },
   },

   // Events
   cancel: $.noop,   // This is the function to be called when the cancel <button> is clicked
   finish: $.noop   // This is the function to be called when the finish <button> is clicked
});
